jQuery vs Vanilla JS
// ── 셀렉터 ──
$('.btn')                     → document.querySelector('.btn')
$('.btn')                     → document.querySelectorAll('.btn')  (전체)
$('#modal')                   → document.getElementById('modal')

// ── 이벤트 ──
$('.btn').on('click', fn)      → el.addEventListener('click', fn)

// ── 스타일/클래스 ──
$('.box').addClass('active')  → el.classList.add('active')
$('.box').css('color', 'red') → el.style.color = 'red'
$('.box').hide()               → el.style.display = 'none'
$('.box').show()               → el.style.display = ''

// ── 내용 ──
$('.title').text('제목')       → el.textContent = '제목'
$('.box').html('<b>내용</b>') → el.innerHTML = '<b>내용</b>'
$('input').val()              → input.value

// ── Ajax → fetch ──
$.ajax({ url: '/api', success: fn }) → fetch('/api').then(r => r.json()).then(fn)

✓ Vanilla JS / React hooks 사용
// jQuery 없이 동일한 기능
document.querySelector('.btn').addEventListener('click', handler);
el.classList.toggle('active');
const val = document.querySelector('input').value;
⚠️ jQuery는 약 30kb 의존성이다. Vanilla JS로 가능한 것을 jQuery로 하는 건 불필요한 번들 크기 증가다. 신규 프로젝트에서 jQuery 추가는 코드 리뷰에서 지적받는다.
