jsx — CSS vs Tailwind 비교
/* CSS 방식 */
const Card = () => (
  <div className="card">
    <h2 className="card-title">제목</h2>
  </div>
);
/* → 별도 .css 파일 필요 */

/* Tailwind 방식 */
const Card = () => (
  <div className="p-6 rounded-lg border border-gray-700 bg-gray-800 hover:border-blue-500 transition-colors">
    <h2 className="text-lg font-bold text-white mb-2">제목</h2>
  </div>
);
/* → CSS 파일 없음 */

✓ CLI 빌드로 사용 클래스만 추출
# npm으로 설치 후 purge 적용
npm install -D tailwindcss
npx tailwindcss init
# 빌드 결과: 사용한 클래스만 ~5kb
⚠️ CDN은 모든 클래스(3MB+)를 내려받는다. 실무/포트폴리오 배포엔 반드시 CLI 빌드. React 프로젝트라면 npx create-react-app 후 Tailwind 설정 가이드 따라가면 된다.
